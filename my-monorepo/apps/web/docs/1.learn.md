## 🧵 React Fiber 架构详解：时间切片与工作循环

---

## 📚 核心概念

### 什么是 Fiber？

**Fiber** 是 React 16 引入的新架构，核心目标是实现**可中断的异步渲染**。

| 传统架构 (Stack Reconciler) | Fiber 架构 |
|---------------------------|-----------|
| 递归调用，同步执行 | 链表结构，可中断 |
| 一旦开始必须完成 | 可暂停、恢复、丢弃 |
| 无法优先级调度 | 支持优先级调度 |
| 阻塞主线程 | 利用浏览器空闲时间 |

---

## 🎯 时间切片（Time Slicing）原理

```javascript
/**
 * 核心思想：将大型任务拆分成小块，在浏览器空闲时执行
 * 
 * 1. 检查剩余时间（deadline.timeRemaining()）
 * 2. 如果时间充足，执行一个工作单元
 * 3. 如果时间不足，暂停并等待下一次空闲
 * 4. 重复直到所有工作完成
 */

let nextUnitOfWork = null;  // 下一个工作单元
let currentRoot = null;      // 当前根节点
let workInProgress = null;   // 正在进行的工作

function workLoop(deadline) {
  // 是否应该让出控制权给浏览器
  let shouldYield = false;
  
  // 只要还有工作单元且时间充足，就继续执行
  while (nextUnitOfWork && !shouldYield) {
    nextUnitOfWork = performUnitOfWork(nextUnitOfWork);
    shouldYield = deadline.timeRemaining() < 1;
  }
  
  // 如果所有工作单元完成，提交更改
  if (!nextUnitOfWork && workInProgress) {
    commitRoot();
  }
  
  // 请求下一次空闲回调
  requestIdleCallback(workLoop);
}

// 启动工作循环
requestIdleCallback(workLoop);
```

## 🔧 完整实现示例

### 1️⃣ Fiber 节点结构

```javascript
/**
 * Fiber 节点的数据结构
 */
function createFiberNode(element, parentFiber) {
  return {
    // DOM 相关
    type: element.type,          // 节点类型（div, span 等）
    props: element.props,        // 节点属性
    dom: null,                   // 对应的 DOM 节点
    
    // 树结构相关
    parent: parentFiber,         // 父节点
    child: null,                 // 第一个子节点
    sibling: null,               // 下一个兄弟节点
    
    // 工作相关
    alternate: null,             // 上一次渲染的 Fiber（用于 diff）
    effectTag: null,             // 副作用标记（PLACEMENT, UPDATE, DELETION）
    effects: [],                 // 需要提交的副作用列表
  };
}
```

### 2️⃣ 执行工作单元（performUnitOfWork）

```javascript
/**
 * 执行一个工作单元
 * 1. 创建 DOM 节点
 * 2. 为子元素创建 Fiber 节点
 * 3. 返回下一个工作单元（深度优先遍历）
 */
function performUnitOfWork(fiber) {
  // 1. 创建 DOM 节点（如果还没有）
  if (!fiber.dom) {
    fiber.dom = createDom(fiber);
  }
  
  // 2. 为子元素创建 Fiber 节点
  const elements = fiber.props.children || [];
  reconcileChildren(fiber, elements);
  
  // 3. 返回下一个工作单元（深度优先）
  // 优先级：子节点 > 兄弟节点 > 叔叔节点
  if (fiber.child) {
    return fiber.child;
  }
  
  let nextFiber = fiber;
  while (nextFiber) {
    if (nextFiber.sibling) {
      return nextFiber.sibling;
    }
    nextFiber = nextFiber.parent;
  }
}

/**
 * 创建 DOM 节点
 */
function createDom(fiber) {
  const dom = fiber.type === 'TEXT_ELEMENT'
    ? document.createTextNode('')
    : document.createElement(fiber.type);
  
  // 设置属性
  updateDomProperties(dom, {}, fiber.props);
  
  return dom;
}

/**
 * 协调子节点（简单版本：每次都重新创建）
 */
function reconcileChildren(fiber, elements) {
  let index = 0;
  let prevSibling = null;
  
  while (index < elements.length) {
    const element = elements[index];
    
    // 创建子 Fiber 节点
    const newFiber = {
      type: element.type,
      props: element.props,
      parent: fiber,
      dom: null,
    };
    
    // 构建链表
    if (index === 0) {
      fiber.child = newFiber;
    } else {
      prevSibling.sibling = newFiber;
    }
    
    prevSibling = newFiber;
    index++;
  }
}
```

### 3️⃣ 提交阶段（commitRoot）

```javascript
/**
 * 提交阶段：将所有更改应用到 DOM
 * 这个阶段必须同步执行，不能中断
 */
function commitRoot() {
  // 提交副作用到 DOM
  commitWork(workInProgress.child);
  
  // 保存当前根节点
  currentRoot = workInProgress;
  
  // 重置工作单元
  workInProgress = null;
}

/**
 * 递归提交工作
 */
function commitWork(fiber) {
  if (!fiber) return;
  
  // 获取父 DOM 节点
  let domParentFiber = fiber.parent;
  while (!domParentFiber.dom) {
    domParentFiber = domParentFiber.parent;
  }
  const domParent = domParentFiber.dom;
  
  // 根据 effectTag 执行操作
  if (fiber.effectTag === 'PLACEMENT' && fiber.dom !== null) {
    domParent.appendChild(fiber.dom);
  }
  
  // 递归提交子节点和兄弟节点
  commitWork(fiber.child);
  commitWork(fiber.sibling);
}
```

### 4️⃣ 渲染函数（render）

```javascript
/**
 * 渲染函数：启动 Fiber 工作循环
 */
function render(element, container) {
  // 创建根 Fiber 节点
  workInProgress = {
    dom: container,
    props: {
      children: [element],
    },
    alternate: currentRoot,
  };
  
  // 设置第一个工作单元
  nextUnitOfWork = workInProgress;
}
```

## 🎮 完整示例：简易 React

```javascript
/**
 * 简易 React Fiber 实现
 */

let nextUnitOfWork = null;
let currentRoot = null;
let workInProgress = null;

// ========== DOM 操作 ==========
function updateDomProperties(dom, prevProps, nextProps) {
  // 移除旧属性
  Object.keys(prevProps).forEach(name => {
    if (name !== 'children' && !(name in nextProps)) {
      dom.removeAttribute(name);
    }
  });
  
  // 设置新属性
  Object.keys(nextProps).forEach(name => {
    if (name !== 'children') {
      dom[name] = nextProps[name];
    }
  });
}

function createDom(fiber) {
  const dom = fiber.type === 'TEXT_ELEMENT'
    ? document.createTextNode('')
    : document.createElement(fiber.type);
  
  updateDomProperties(dom, {}, fiber.props);
  return dom;
}

// ========== 协调子节点 ==========
function reconcileChildren(fiber, elements) {
  let index = 0;
  let prevSibling = null;
  
  while (index < elements.length) {
    const element = elements[index];
    
    const newFiber = {
      type: element.type,
      props: element.props,
      parent: fiber,
      dom: null,
      effectTag: 'PLACEMENT',  // 新节点需要插入
    };
    
    if (index === 0) {
      fiber.child = newFiber;
    } else {
      prevSibling.sibling = newFiber;
    }
    
    prevSibling = newFiber;
    index++;
  }
}

// ========== 执行工作单元 ==========
function performUnitOfWork(fiber) {
  // 1. 创建 DOM
  if (!fiber.dom) {
    fiber.dom = createDom(fiber);
  }
  
  // 2. 协调子节点
  const elements = fiber.props.children || [];
  reconcileChildren(fiber, elements);
  
  // 3. 返回下一个工作单元（深度优先）
  if (fiber.child) {
    return fiber.child;
  }
  
  let nextFiber = fiber;
  while (nextFiber) {
    if (nextFiber.sibling) {
      return nextFiber.sibling;
    }
    nextFiber = nextFiber.parent;
  }
}

// ========== 提交阶段 ==========
function commitWork(fiber) {
  if (!fiber) return;
  
  // 找到父 DOM 节点
  let domParentFiber = fiber.parent;
  while (!domParentFiber.dom) {
    domParentFiber = domParentFiber.parent;
  }
  const domParent = domParentFiber.dom;
  
  // 插入 DOM
  if (fiber.effectTag === 'PLACEMENT' && fiber.dom !== null) {
    domParent.appendChild(fiber.dom);
  }
  
  // 递归提交
  commitWork(fiber.child);
  commitWork(fiber.sibling);
}

function commitRoot() {
  commitWork(workInProgress.child);
  currentRoot = workInProgress;
  workInProgress = null;
}

// ========== 工作循环 ==========
function workLoop(deadline) {
  let shouldYield = false;
  
  while (nextUnitOfWork && !shouldYield) {
    nextUnitOfWork = performUnitOfWork(nextUnitOfWork);
    shouldYield = deadline.timeRemaining() < 1;
  }
  
  if (!nextUnitOfWork && workInProgress) {
    commitRoot();
  }
  
  requestIdleCallback(workLoop);
}

// 启动工作循环
requestIdleCallback(workLoop);

// ========== 渲染函数 ==========
function render(element, container) {
  workInProgress = {
    dom: container,
    props: {
      children: [element],
    },
    alternate: currentRoot,
  };
  
  nextUnitOfWork = workInProgress;
}

// ========== JSX 兼容 ==========
function createElement(type, props, ...children) {
  return {
    type,
    props: {
      ...props,
      children: children.map(child =>
        typeof child === 'object' ? child : createTextElement(child)
      ),
    },
  };
}

function createTextElement(text) {
  return {
    type: 'TEXT_ELEMENT',
    props: {
      nodeValue: text,
      children: [],
    },
  };
}

// ========== 导出 ==========
const Didact = {
  createElement,
  render,
};

/** @jsx Didact.createElement */

// 使用示例
function App() {
  return (
    <div>
      <h1>Hello Fiber!</h1>
      <p>This is a simple React Fiber implementation</p>
    </div>
  );
}

Didact.render(<App />, document.getElementById('root'));
```

## 📊 Fiber 遍历过程可视化

```
        Root
         │
         ▼
       [A] ← nextUnitOfWork
      ╱  ╲
     ▼    ▼
   [B]   [C]
   ╱ ╲
  ▼   ▼
[D]  [E]

执行顺序：
1. A → 创建 DOM，协调子节点
2. B → 创建 DOM，协调子节点
3. D → 创建 DOM，无子节点
4. E → 创建 DOM，无子节点
5. C → 创建 DOM，无子节点
```

## 🎯 关键要点总结

| 概念 | 说明 |
|------|------|
| **工作单元** | 一个 Fiber 节点，代表一小块工作 |
| **时间切片** | 利用 `requestIdleCallback` 在空闲时执行 |
| **可中断** | 检查 `deadline.timeRemaining()` 决定是否暂停 |
| **链表结构** | 通过 `child`/`sibling`/`parent` 链接，替代递归 |
| **提交阶段** | 必须同步执行，将所有更改一次性应用到 DOM |
| **优先级** | 可根据任务类型分配不同优先级 |

---

## ⚠️ 注意事项

1. **`requestIdleCallback` 兼容性**：现代浏览器支持，但 React 使用自己的调度器
2. **优先级调度**：真实 React 有多种优先级（Immediate, UserBlocking, Normal 等）
3. **Diff 算法**：示例中简化了，真实实现会复用节点
4. **Hooks**：Fiber 架构支持 Hooks，每个 Hook 也是 Fiber 节点
